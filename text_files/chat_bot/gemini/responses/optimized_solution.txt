```python
# 10.2 Write a program to read through the mbox-short.txt
# and figure out the distribution by hour of the day for each
# of the messages. You can pull the hour out from
# the 'From ' line by finding the time and then splitting the string
# a second time using a colon.
# From stephen.marquard@uct.ac.za Sat Jan  5 09:14:16 2008
# Once you have accumulated the
# counts for each hour, print out the counts,
# sorted by hour as shown below.

# Summary of changes:
# 1.  Combined Loops: The code now processes each line and updates the hour count
#     in a single pass through the file. This eliminates the need for the
#     intermediate list `times_emails_sent`, making the program more memory-efficient.
# 2.  Simplified Logic: Instead of storing full time-stamps and processing them later,
#     the hour is extracted and counted immediately. The line parsing logic was also
#     made more direct.
# 3.  Efficient Sorting and Printing: The final loop to create a list of tuples for
#     sorting was removed. The optimized code now directly sorts the dictionary's
#     .items() and prints them in a single, more Pythonic loop.

# Initialize the dictionary to store hour counts directly.
hours_distribution = dict()

while True:
    try:
        file_name = input('Please enter a text file: ')
        if len(file_name) < 1:
            file_name = 'mbox-short.txt'
        with open(file_name) as file_handle:
            for line in file_handle:
                # Optimized line check: 'From ' with a space is more specific and reliable.
                if line.startswith('From '):
                    # Split the line once and directly access the time and then the hour.
                    words = line.split()
                    # The time string '09:14:16' is the 6th word (index 5).
                    time = words[5]
                    # Get the hour by splitting the time string on the colon.
                    hour = time.split(':')[0]
                    # Populate the dictionary in the same loop, eliminating the intermediate list.
                    hours_distribution[hour] = hours_distribution.get(hour, 0) + 1
        # Break the loop once the file is successfully processed.
        break

    except FileNotFoundError:
        print('The file you entered does not exist. Please enter a valid file name.')


# Use a single loop to sort and print the results.
# sorted() on a dictionary's .items() view returns a sorted list of (key, value) tuples.
for hour, count in sorted(hours_distribution.items()):
    # Unpack the tuple directly in the for loop and print.
    print(hour, count)
```
